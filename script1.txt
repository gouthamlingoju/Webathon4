import json
from datetime import datetime

class Task:
    def __init__(self, title, priority):
        self.title = title
        self.priority = priority
        self.created = datetime.now().isoformat()
        self.completed = False

    def mark_done(self):
        self.completed = True

    def to_dict(self):
        return {
            "title": self.title,
            "priority": self.priority,
            "created": self.created,
            "completed": self.completed
        }

class TaskManager:
    def __init__(self):
        self.tasks = []

    def add_task(self, title, priority):
        self.tasks.append(Task(title, priority))

    def complete_task(self, index):
        if 0 <= index < len(self.tasks):
            self.tasks[index].mark_done()

    def sort_tasks(self):
        self.tasks.sort(key=lambda t: (-t.priority, t.completed))

    def display(self):
        for i, t in enumerate(self.tasks):
            status = "✔" if t.completed else "✘"
            print(f"{i}. [{status}] {t.title} (Priority {t.priority})")

    def save(self, filename="tasks.json"):
        with open(filename, "w") as f:
            json.dump([t.to_dict() for t in self.tasks], f, indent=4)

def main():
    tm = TaskManager()
    tm.add_task("Finish report", 3)
    tm.add_task("Workout", 2)
    tm.add_task("Read book", 1)

    tm.complete_task(1)
    tm.sort_tasks()
    tm.display()
    tm.save()

if __name__ == "__main__":
    main()
